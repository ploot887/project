public class GuiUtils {
    public static void drawGlow(DrawContext context, int x, int y, int w, int h, int color) {
        // Отрисовка "свечения" вокруг панели (просто цветной прямоугольник)
        int a = (color >> 24 & 255);
        int r = (color >> 16 & 255);
        int g = (color >> 8 & 255);
        int b = (color & 255);
        context.fill(x, y, x + w, y + h, color);
    }

    public static void drawDarkPanel(DrawContext context, int x, int y, int w, int h) {
        // Отрисовка темной панели, как в Blockbuster
        drawGlow(context, x, y, w, h, 0xdd000000);
    }
}

package seq.sequencermod.client.gui.panel;

import net.minecraft.client.MinecraftClient;
import net.minecraft.client.gui.DrawContext;
import net.minecraft.client.gui.widget.TextFieldWidget;
import net.minecraft.text.Text;
import seq.sequencermod.client.gui.GuiUtils;
import seq.sequencermod.client.gui.widget.panel.GuiMorphPanel;
import seq.sequencermod.morph.PlayerMorph;

public class GuiPlayerMorphPanel extends GuiMorphPanel {

    private TextFieldWidget nameField;
    private final PlayerMorph playerMorph;

    public GuiPlayerMorphPanel(MinecraftClient client, PlayerMorph morph) {
        super(client, morph);
        this.playerMorph = morph;
    }

    @Override
    public void open() {
        super.open();
        this.nameField = new TextFieldWidget(this.client.textRenderer, 0, 0, 100, 20, Text.literal(""));
        this.nameField.setText(this.playerMorph.name);
        this.nameField.setFocused(true);
    }

    @Override
    public void close() {
        super.close();
        if (this.nameField != null) {
            this.playerMorph.name = this.nameField.getText();
        }
    }

    @Override
    public void renderButton(DrawContext context, int mouseX, int mouseY, float delta) {
        // Позиционируем текстовое поле здесь, чтобы оно всегда было в правильном месте
        if (this.nameField != null) {
            this.nameField.setX(this.getX() + 10);
            this.nameField.setY(this.getY() + 10);
        }

        // Отрисовка области предпросмотра
        int previewX = this.getX() + this.width - 110;
        int previewY = this.getY() + 10;
        int previewW = 100;
        int previewH = 150;
        GuiUtils.drawDarkPanel(context, previewX, previewY, previewW, previewH);
        context.drawCenteredTextWithShadow(this.client.textRenderer, "Preview", previewX + previewW / 2, previewY + 5, 0xaaaaaa);

        // Отрисовка текстового поля
        if (this.nameField != null) {
            this.nameField.render(context, mouseX, mouseY, delta);
        }
    }

    @Override
    public boolean mouseClicked(double mouseX, double mouseY, int button) {
        if (this.nameField != null) {
            return this.nameField.mouseClicked(mouseX, mouseY, button);
        }
        return super.mouseClicked(mouseX, mouseY, button);
    }

    @Override
    public boolean keyPressed(int keyCode, int scanCode, int modifiers) {
        if (this.nameField != null && this.nameField.isFocused()) {
            return this.nameField.keyPressed(keyCode, scanCode, modifiers);
        }
        return super.keyPressed(keyCode, scanCode, modifiers);
    }

    @Override
    public boolean charTyped(char chr, int modifiers) {
        if (this.nameField != null && this.nameField.isFocused()) {
            return this.nameField.charTyped(chr, modifiers);
        }
        return super.charTyped(chr, modifiers);
    }
}

package seq.sequencermod.client.gui.screen;

import net.minecraft.client.gui.DrawContext;
import net.minecraft.client.gui.screen.Screen;
import net.minecraft.client.gui.widget.ButtonWidget;
import net.minecraft.text.Text;
import seq.sequencermod.client.gui.MorphPanelFactory;
import seq.sequencermod.client.gui.widget.panel.GuiMorphPanel;
import seq.sequencermod.morph.AbstractMorph;
import seq.sequencermod.morph.PlayerMorph;

import java.util.ArrayList;
import java.util.List;

public class MorphScreen extends Screen {

    private final List<AbstractMorph> morphs = new ArrayList<>();
    private AbstractMorph selectedMorph;
    private GuiMorphPanel currentPanel;

    public MorphScreen() {
        super(Text.literal("Morph Screen"));
        // Для теста добавим один морф игрока
        this.morphs.add(new PlayerMorph());
    }

    @Override
    protected void init() {
        super.init();
        // Кнопка для закрытия экрана
        this.addDrawableChild(ButtonWidget.builder(Text.literal("Done"), (button) -> {
            this.close();
        }).dimensions(this.width / 2 - 50, this.height - 30, 100, 20).build());

        // Выбираем первый морф по умолчанию
        if (!this.morphs.isEmpty()) {
            this.selectMorph(this.morphs.get(0));
        }
    }

    private void selectMorph(AbstractMorph morph) {
        if (this.currentPanel != null) {
            this.currentPanel.close();
            // В новых версиях метод remove, в старых может быть removeSelectable
            this.remove(this.currentPanel);
        }

        this.selectedMorph = morph;
        // Здесь мы используем нашу фабрику!
        this.currentPanel = MorphPanelFactory.createPanelFor(this.selectedMorph, this.client);

        if (this.currentPanel != null) {
            // Размещаем панель
            this.currentPanel.setX(130); // Отступ слева для списка
            this.currentPanel.setY(20);
            // ИСПОЛЬЗУЕМ СЕТТЕРЫ
            this.currentPanel.setWidth(this.width - 140);
            this.currentPanel.setHeight(this.height - 50);
            this.currentPanel.open();
            this.addSelectableChild(this.currentPanel);
        }
    }

    @Override
    public void render(DrawContext context, int mouseX, int mouseY, float delta) {
        // Фон - ИСПРАВЛЕННАЯ ВЕРСИЯ
        this.renderBackground(context);

        // Отрисовка списка морфов (пока простой текст)
        context.drawTextWithShadow(this.textRenderer, "Morphs", 10, 10, 0xFFFFFF);
        for (int i = 0; i < this.morphs.size(); i++) {
            AbstractMorph morph = this.morphs.get(i);
            int color = (morph == this.selectedMorph) ? 0xFFFF55 : 0xFFFFFF; // Желтый для выбранного
            context.drawTextWithShadow(this.textRenderer, morph.name, 20, 30 + i * 12, color);
        }

        // Отрисовка панели и дочерних виджетов
        super.render(context, mouseX, mouseY, delta);
    }

    @Override
    public void close() {
        // Убедимся, что панель сохраняет свои данные при закрытии
        if (this.currentPanel != null) {
            this.currentPanel.close();
        }
        super.close();
    }
}

package seq.sequencermod.client.screen;

import net.minecraft.client.gui.DrawContext;
import net.minecraft.client.gui.screen.ingame.HandledScreen;
import net.minecraft.entity.player.PlayerInventory;
import net.minecraft.text.Text;
import seq.sequencermod.client.gui.MorphPanelFactory;
import seq.sequencermod.client.gui.widget.panel.GuiMorphPanel;
import seq.sequencermod.handler.SequencerScreenHandler;
import seq.sequencermod.morph.AbstractMorph;
import seq.sequencermod.morph.PlayerMorph;

import java.util.ArrayList;
import java.util.List;

public class SequencerScreen extends HandledScreen<SequencerScreenHandler> {

    private final List<AbstractMorph> morphs = new ArrayList<>();
    private AbstractMorph selectedMorph;
    private GuiMorphPanel currentPanel;

    // ИСПРАВЛЕНИЕ: Вот правильный конструктор, который нужен для регистрации
    public SequencerScreen(SequencerScreenHandler handler, PlayerInventory inventory, Text title) {
        super(handler, inventory, title);
        this.backgroundWidth = 300;
        this.backgroundHeight = 200;

        // Для теста добавим один морф игрока
        this.morphs.add(new PlayerMorph());
    }

    @Override
    protected void init() {
        super.init();
        if (!this.morphs.isEmpty()) {
            this.selectMorph(this.morphs.get(0));
        }
    }

    private void selectMorph(AbstractMorph morph) {
        if (this.currentPanel != null) {
            this.currentPanel.close();
            this.remove(this.currentPanel);
        }

        this.selectedMorph = morph;
        this.currentPanel = MorphPanelFactory.createPanelFor(this.selectedMorph, this.client);

        // Пусто, так как мы рисуем все в render()

import net.minecraft.block.BlockEntityProvider;

import net.minecraft.block.BlockWithEntity;
import net.minecraft.block.entity.BlockEntity;
import net.minecraft.entity.player.PlayerEntity;

import net.minecraft.util.ActionResult;
import net.minecraft.util.Hand;

import net.minecraft.util.math.BlockPos;
import net.minecraft.world.World;
import org.jetbrains.annotations.Nullable;
import seq.sequencermod.block.entity.SequencerBlockEntity;

public class SequencerBlock extends BlockWithEntity implements BlockEntityProvider {

    public SequencerBlock(Settings settings) {
        super(settings);
    }

    @Override
    public BlockRenderType getRenderType(BlockState state) {
        return BlockRenderType.MODEL;
    }

    @Nullable
    @Override
    public BlockEntity createBlockEntity(BlockPos pos, BlockState state) {
        return new SequencerBlockEntity(pos, state);
    }

    @Override
    public ActionResult onUse(BlockState state, World world, BlockPos pos, PlayerEntity player, Hand hand, BlockHitResult hit) {
        if (!world.isClient) {
            // Этот вызов теперь работает с ExtendedScreenHandlerFactory
            NamedScreenHandlerFactory screenHandlerFactory = state.createScreenHandlerFactory(world, pos);
            if (screenHandlerFactory != null) {
                player.openHandledScreen(screenHandlerFactory);
            }
        }
        return ActionResult.SUCCESS;
    }
}

    private final AnimatableInstanceCache cache = GeckoLibUtil.createInstanceCache(this);

    public GeckoEntity(EntityType<? extends PathAwareEntity> entityType, World world) {
        super(entityType, world);
    }

    public static DefaultAttributeContainer.Builder setAttributes() {
        return PathAwareEntity.createMobAttributes()
                .add(EntityAttributes.GENERIC_MAX_HEALTH, 10.0D)
                .add(EntityAttributes.GENERIC_MOVEMENT_SPEED, 0.3D);
    }

    @Override
    protected void initGoals() {
        this.goalSelector.add(1, new SwimGoal(this));
        this.goalSelector.add(2, new WanderAroundFarGoal(this, 0.75f, 1));
        this.goalSelector.add(3, new LookAtEntityGoal(this, PlayerEntity.class, 8.0f));
        this.goalSelector.add(4, new LookAroundGoal(this));
    }

    @Override
    public void registerControllers(AnimatableManager.ControllerRegistrar controllers) {
        controllers.add(new AnimationController<>(this, "controller", 0, this::predicate));
    }

    // --- ИЗМЕНЕННЫЙ МЕТОД ---
    private PlayState predicate(AnimationState<GeckoEntity> state) {
        // Если сущность движется...
        if (state.isMoving()) {
            // ...проигрываем анимацию 'walk' в цикле.
            return state.setAndContinue(RawAnimation.begin().thenLoop("walk"));
        }
        // Если стоит на месте...
        // ...проигрываем анимацию 'idle' в цикле.
        return state.setAndContinue(RawAnimation.begin().thenLoop("idle"));
    }
    // -------------------------

    @Override
    public AnimatableInstanceCache getAnimatableInstanceCache() {
        return this.cache;
    }
}

package seq.sequencermod.morph;

import net.minecraft.nbt.NbtCompound;

public abstract class AbstractMorph {
    public String name;
    public boolean hasCustomName;

    public AbstractMorph() {
        this.name = "";
        this.hasCustomName = false;
    }

    public void fromNbt(NbtCompound nbt) {
        if (nbt.contains("Name")) {
            this.name = nbt.getString("Name");
        }
    }

    public void toNbt(NbtCompound nbt) {
        if (this.name != null && !this.name.isEmpty()) {
            nbt.putString("Name", this.name);
        }
    }

    public abstract AbstractMorph copy();
}
